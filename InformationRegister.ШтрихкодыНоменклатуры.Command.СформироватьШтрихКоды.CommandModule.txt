
// Создает все отсутствующие ШК для номенклатуры
//
// Параметры
//  Параметр  - Параметр команды - для чего создаем ШК
//
// Возвращаемое значение:
//   Число   - количество созданных ШК
//
&НаСервере
Функция СоздатьШтрихкодыСервер(Номенклатура)
	
	Возврат РегистрыСведений.ШтрихкодыНоменклатуры.СформироватьШтучныеШтрихкодыEAN13(Номенклатура);
	
КонецФункции // СоздатьШтрихкодыСервер()

&НаКлиенте
Процедура ОбработкаКоманды(ПараметрКоманды, ПараметрыВыполненияКоманды)
	
	Если ТипЗнч(ПараметрКоманды) = Тип("СправочникСсылка.Номенклатура") Тогда
		
		Если ОбщегоНазначенияДУМ.ЗначениеРеквизитаОбъекта(ПараметрКоманды, "ЭтоГруппа") Тогда
			ТекстПредупреждения = НСтр("ru = 'Штрих-коды не могут быть сформированы для группы номенклатуры'");
			Предупреждение(ТекстПредупреждения);
			Возврат;
		КонецЕсли;
		
	КонецЕсли;
	
	КоличестовСозданныхШК = СоздатьШтрихкодыСервер(ПараметрКоманды);
	
	Если КоличестовСозданныхШК = -1 Тогда
		
		ТекстПредупреждения = НСтр("ru = 'Исчерпан запас штучных штрих-кодов EAN13'");
		Предупреждение(ТекстПредупреждения);
		
	ИначеЕсли КоличестовСозданныхШК = 0 Тогда
		
		ТекстПредупреждения = НСтр("ru = 'Сформировано штрих-кодов: 0. Для номенклатуры ""%Номенклатура%"" уже введены штрих-коды.'");
		ТекстПредупреждения = СтрЗаменить(ТекстПредупреждения, "%Номенклатура%", ПараметрКоманды);
		
		Предупреждение(ТекстПредупреждения);
		
	Иначе
		
		ТекстПредупреждения = НСтр("ru = 'Сформировано штрих-кодов: %ЧислоШтрихкодов%.'");
		ОбщегоНазначенияКлиентСервер.СообщитьПользователю(СтрЗаменить(ТекстПредупреждения, "%ЧислоШтрихкодов%", КоличестовСозданныхШК));
		
		ОповеститьОбИзменении(Тип("СправочникСсылка.ХарактеристикиНоменклатуры"));
		ОповеститьОбИзменении(Тип("РегистрСведенийКлючЗаписи.ШтрихкодыНоменклатуры"));
		
	КонецЕсли;
	
КонецПроцедуры
