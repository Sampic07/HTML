	    НеДостача = Истина;

		Стр.Количество = 0;
		Стр.Сумма = 0;
		ТоварТовар = Истина;

		ТоварТовар = Истина;

		Для каждого тДерово Из гДерево.ПолучитьЭлементы() Цикл
